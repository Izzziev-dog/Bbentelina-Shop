        function updateCartCount() {

            const count = cart.reduce((total, item) => total + item.quantity, 0);

            cartCount.textContent = count;

        // Функция обновления суммы корзины

        function updateCartTotal() {

            const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);

            cartTotalPrice.textContent = total.toLocaleString('ru-RU') + ' ₽';

        }

                        <div class="cart-item-price">${item.price.toLocaleString('ru-RU')} ₽ × ${item.quantity}</div>

                    </div>

                    <button class="cart-item-remove" data-id="${item.id}">×</button>

                `;

                cartItems.appendChild(cartItem);

            });

                });

        }

        document.querySelectorAll('.add-to-cart').forEach(button => {

                const id = productCard.getAttribute('data-id');

                const name = productCard.getAttribute('data-name');

                const price = parseInt(productCard.getAttribute('data-price'));

                const image = productCard.querySelector('img').src;

                

                // Создаем эффект волны

                const ripple = document.createElement('span');

                ripple.classList.add('ripple');

                this.appendChild(ripple);

                

                setTimeout(() => {

                    ripple.remove();

                }, 600);

                

                // Проверяем, есть ли уже такой товар в корзине

                const existingItem = cart.find(item => item.id === id);

                if (existingItem) {

                    existingItem.quantity += 1;

                } else {

                    cart.push({

                        id,

                        name,

                        price,

                        quantity: 1,

                        image

                    });

                }

                updateCartCount();

                

                // Меняем текст кнопки

                this.classList.add('added');

                

                setTimeout(() => {

                    this.textContent = 'В корзину';

                    this.classList.remove('added');

                }, 2000);

            });

        });

 

        // Удаление товара из корзины

        function removeFromCart(id) {

            cart = cart.filter(item => item.id !== id);

            renderCartItems();

            updateCartCount();

            updateCartTotal();
